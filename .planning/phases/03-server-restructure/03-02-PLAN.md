From server/emailFinder.js lines 347-358 (sequential probe loop to parallelize):
```javascript
// CURRENT (sequential — replace this):
const probed = [];
for (const candidate of candidates) {
  let smtpResult;
  if (probedEmails.has(candidate.email)) {
    smtpResult = probedEmails.get(candidate.email);
  } else {
    smtpResult = await smtpProbe(mxHost, candidate.email);
    probedEmails.set(candidate.email, smtpResult);
  }
  probed.push({ ...candidate, smtpResult });
}

// REPLACEMENT (parallel):
const probeResults = await Promise.allSettled(
  candidates.map(async (candidate) => {
    let smtpResult;
    if (probedEmails.has(candidate.email)) {
      smtpResult = probedEmails.get(candidate.email);
    } else {
      smtpResult = await smtpProbe(mxHost, candidate.email);
      probedEmails.set(candidate.email, smtpResult);
    }
    return { ...candidate, smtpResult };
  })
);
const probed = probeResults
  .filter(r => r.status === 'fulfilled')
  .map(r => r.value);
```

Note: smtpProbe() always resolves (never rejects). The .filter(fulfilled) is defensive but safe. The catch-all probe above the loop (lines 337-345) stays sequential — it's a single probe.
