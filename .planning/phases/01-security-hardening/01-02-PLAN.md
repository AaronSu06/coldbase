**server/index.js changes:**

2. Replace `app.use(cors())` (line 13) with origin-allowlist CORS:
```js
const allowedOrigins = process.env.ALLOWED_ORIGINS
  ? process.env.ALLOWED_ORIGINS.split(',').map(o => o.trim())
  : ['http://localhost:5173'];

app.use(cors({
  origin: (origin, callback) => {
    // Allow requests with no origin (curl, server-to-server) and chrome-extension:// origins
    if (!origin || origin.startsWith('chrome-extension://') || allowedOrigins.includes(origin)) {
      callback(null, true);
    } else {
      callback(new Error('CORS: origin not allowed'));
    }
  },
  credentials: true,
}));
```

3. Fix the `requireSecret` middleware to reject when env var is absent (not skip the check):
```js
function requireSecret(req, res, next) {
  const secret = process.env.REACH_SECRET;
  if (!secret) {
    return res.status(500).json({ error: 'Server misconfigured: REACH_SECRET not set' });
  }
  if (req.headers['x-reach-secret'] !== secret) {
    return res.status(401).json({ error: 'Unauthorized', message: 'Missing or invalid x-reach-secret header' });
  }
  next();
}
```

4. Apply `requireSecret` to all /api/* routes via middleware. Add this line AFTER `app.use(express.json())` and BEFORE any route definitions:
```js
app.use('/api', requireSecret);
```

The `GET /track/:trackingId` route is NOT under `/api/` path — it stays public by design.

5. Create the rate limiter and apply it to the three expensive routes. Add after the requireSecret registration:
```js
const expensiveRateLimit = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutes
  max: 10,
  standardHeaders: true,
  legacyHeaders: false,
  message: { error: 'Too many requests' },
});
```

Then apply per-route on the expensive endpoints. Find the route definitions for `/api/find-email`, `/api/draft-email`, `/api/suggest-domains` and add the limiter as middleware:
```js
app.post('/api/find-email', expensiveRateLimit, async (req, res) => { ... });
app.post('/api/draft-email', expensiveRateLimit, async (req, res) => { ... });
app.post('/api/suggest-domains', expensiveRateLimit, async (req, res) => { ... });
```

**Middleware ordering in index.js must be:**

2. `app.use(express.json())` — body parsing
3. `app.use('/api', requireSecret)` — auth gate for all /api routes
